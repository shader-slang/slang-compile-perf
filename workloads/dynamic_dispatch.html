[shader("compute")]
[numthreads(1,1,1)]
void computeMain(uint3 tid : SV_DispatchThreadID)
{
    float acc = 0.0;
    int base = int(tid.x);
    for (int i = 0; i < 200; ++i)
        acc += dispatch((base + i) % 200, outBuf[0] + float(i));
    outBuf[0] = acc;
}
