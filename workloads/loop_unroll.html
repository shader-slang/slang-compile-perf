RWStructuredBuffer<float> outBuf;

[shader("compute")]
[numthreads(1,1,1)]
void computeMain(uint3 tid : SV_DispatchThreadID)
{
    float acc = outBuf[tid.x];
    [ForceUnroll] for (int i = 0; i < 300; ++i)
        acc = acc * 1.0009 + sin(acc + float(i)) * 0.5 - cos(acc * 0.5);
    outBuf[0] = acc;
